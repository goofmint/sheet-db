### 1. 認証ミドルウェア (`src/middleware/auth.ts`)

```typescript
import type { Context, MiddlewareHandler } from 'hono';
import { getCookie } from 'hono/cookie';
import type { Env } from '../types/env';
import { SessionService } from '../services/session';
import { ConfigService } from '../services/config';
import { ACLService } from '../services/acl';

/**
 * Authentication context that will be attached to Hono context
 */
export interface AuthContext {
  // User authentication
  userId?: string;
  sessionId?: string;
  isAuthenticated: boolean;
  
  // Master key authentication
  isMasterKey: boolean;
  
  // User roles (cached per request)
  roles?: string[];
}

/**
 * Authentication middleware options
 */
export interface AuthOptions {
  // Whether authentication is required (default: true)
  required?: boolean;
  
  // Allow master key authentication (default: true)
  allowMasterKey?: boolean;
  
  // Custom error response
  onError?: (c: Context) => Response | Promise<Response>;
}

/**
 * Create authentication middleware
 */
export function auth(options: AuthOptions = {}): MiddlewareHandler<{ Bindings: Env }> {
  const {
    required = true,
    allowMasterKey = true,
    onError = defaultErrorResponse
  } = options;

  return async (c, next) => {
    const authContext: AuthContext = {
      isAuthenticated: false,
      isMasterKey: false
    };

    // Attach auth context to Hono context
    c.set('auth', authContext);

    try {
      // 1. Check master key authentication first
      if (allowMasterKey) {
        const masterKey = c.req.header('x-master-key');
        if (masterKey) {
          const isValid = await ACLService.validateMasterKey(masterKey);
          if (isValid) {
            authContext.isAuthenticated = true;
            authContext.isMasterKey = true;
            return await next();
          }
          // Invalid master key - continue to session auth
        }
      }

      // 2. Check session cookie authentication
      const sessionId = getCookie(c, 'session');
      if (sessionId) {
        const validationResult = await SessionService.validateSession(sessionId);
        if (validationResult.success) {
          authContext.isAuthenticated = true;
          authContext.sessionId = sessionId;
          authContext.userId = validationResult.auth0UserId;
          
          // Pre-fetch user roles for this request
          const roles = await ACLService.getUserRoles(authContext.userId, c);
          authContext.roles = roles;
        }
      }

      // 3. Check if authentication is required
      if (required && !authContext.isAuthenticated) {
        return await onError(c);
      }

      await next();
    } catch (error) {
      console.error('Authentication middleware error:', error);
      if (required) {
        return await onError(c);
      }
      // If auth is optional, continue even with errors
      await next();
    }
  };
}

/**
 * Default error response for unauthenticated requests
 */
function defaultErrorResponse(c: Context): Response {
  return c.json({
    success: false,
    error: 'unauthorized',
    message: 'Authentication required'
  }, 401);
}

/**
 * Helper middleware to require specific roles
 */
export function requireRoles(roles: string[]): MiddlewareHandler<{ Bindings: Env }> {
  return async (c, next) => {
    const authContext = c.get('auth') as AuthContext;

    // Master key bypasses role checks
    if (authContext.isMasterKey) {
      return await next();
    }

    if (!authContext.isAuthenticated) {
      return c.json({
        success: false,
        error: 'unauthorized',
        message: 'Authentication required'
      }, 401);
    }

    // Check if user has required roles
    const userRoles = authContext.roles || [];
    const hasRequiredRoles = roles.every(role => userRoles.includes(role));

    if (!hasRequiredRoles) {
      return c.json({
        success: false,
        error: 'forbidden',
        message: 'Insufficient permissions'
      }, 403);
    }

    await next();
  };
}

/**
 * Get authentication context from Hono context
 */
export function getAuth(c: Context): AuthContext {
  return c.get('auth') as AuthContext || {
    isAuthenticated: false,
    isMasterKey: false
  };
}
```

### 使用例

#### 1. 基本的な認証が必要なエンドポイント
```typescript
import { Hono } from 'hono';
import { auth } from '../middleware/auth';

const app = new Hono();

// 認証が必要なエンドポイント
app.use('/api/v1/protected/*', auth());

app.get('/api/v1/protected/data', (c) => {
  const authContext = c.get('auth');
  return c.json({
    userId: authContext.userId,
    isMasterKey: authContext.isMasterKey
  });
});
```

#### 2. 認証がオプショナルなエンドポイント
```typescript
// 認証はオプショナル（認証されていればユーザー情報を使用）
app.use('/api/v1/public/*', auth({ required: false }));

app.get('/api/v1/public/data', (c) => {
  const authContext = c.get('auth');
  if (authContext.isAuthenticated) {
    // 認証済みユーザー向けの追加情報
    return c.json({ 
      data: 'public data',
      personalizedData: `Hello, ${authContext.userId}`
    });
  }
  // 公開データのみ
  return c.json({ data: 'public data' });
});
```

#### 3. 特定のロールが必要なエンドポイント
```typescript
import { auth, requireRoles } from '../middleware/auth';

// 管理者ロールが必要
app.use('/api/v1/admin/*', auth(), requireRoles(['admin']));

app.get('/api/v1/admin/users', (c) => {
  // adminロールを持つユーザーのみアクセス可能
  return c.json({ users: [] });
});
```

#### 4. マスターキー認証を無効化
```typescript
// マスターキー認証を無効化（セッション認証のみ）
app.use('/api/v1/user/*', auth({ allowMasterKey: false }));
```

#### 5. カスタムエラーレスポンス
```typescript
app.use('/api/v1/custom/*', auth({
  onError: (c) => {
    return c.json({
      error: 'Please login to continue',
      loginUrl: '/auth/login'
    }, 401);
  }
}));
```

## 実装時の注意点

1. **パフォーマンス考慮**
   - ユーザーロールはリクエストごとに1回だけ取得（c.set()でキャッシュ）
   - マスターキー検証は高速なハッシュ比較

2. **セキュリティ考慮**
   - タイミング攻撃対策（定数時間比較）
   - エラー時の情報漏洩防止

3. **柔軟性**
   - 認証の必須/オプショナル設定
   - マスターキー認証の有効/無効設定
   - カスタムエラーレスポンス

4. **互換性**
   - 既存のAPIエンドポイントに簡単に追加可能
   - Honoの標準的なミドルウェアパターンに準拠
