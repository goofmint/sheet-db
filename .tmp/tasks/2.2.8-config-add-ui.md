## 概要

既存の設定管理UIに設定項目の追加機能を実装する。POST /api/v1/configs APIを活用してクライアントサイドから新しい設定項目を追加できるようにする。

## 要件

### 機能要件
- 設定一覧画面に「設定を追加」ボタンを追加
- 設定追加用のモーダルダイアログを表示
- 入力項目：key、value、type、description
- クライアントサイドでのバリデーション
- POST /api/v1/configs APIへのリクエスト送信
- 成功時に設定一覧の再読み込み
- エラーハンドリング
- 画面、コード、コメントはすべて英語で記述

### UI要件
- 現在の認証フローを維持（パスワード入力 → 設定表示）
- 追加ボタンは設定一覧表示後に表示
- フォーム項目は明確にラベル付け
- 送信中はローディング状態を表示
- 成功・失敗メッセージの表示

### 技術要件
- 現在のクライアントサイド認証方式を使用
- Authorization Bearerヘッダーでの認証
- 設定タイプ（string、boolean、number、json）の選択機能
- 入力値の適切な型変換

### 2. JavaScriptの実装

#### public/statics/config/client.js に追加する関数

```javascript
// モーダル表示・非表示
function showAddConfigModal() {
  document.getElementById('add-config-modal').style.display = 'block';
}

function hideAddConfigModal() {
  document.getElementById('add-config-modal').style.display = 'none';
  document.getElementById('add-config-form').reset();
}

// 設定追加処理
async function handleAddConfig(event) {
  event.preventDefault();
  
  const formData = new FormData(event.target);
  const configData = {
    key: formData.get('key').trim(),
    value: convertValueByType(formData.get('value'), formData.get('type')),
    type: formData.get('type'),
    description: formData.get('description') || ''
  };
  
  // バリデーション
  if (configData.key === '' || formData.get('value') === '') {
    showErrorInModal('キーと値は必須です。');
    return;
  }
  
  try {
    const submitBtn = event.target.querySelector('button[type="submit"]');
    submitBtn.disabled = true;
    submitBtn.textContent = '追加中...';
    
    const response = await fetch('/api/v1/configs', {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${configPassword}`,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify(configData)
    });
    
    if (!response.ok) {
      const error = await response.json();
      throw new Error(error.error?.message || `HTTP ${response.status}`);
    }
    
    // 成功時の処理
    hideAddConfigModal();
    showSuccessMessage('設定項目を追加しました。');
    
    // 設定一覧を再読み込み
    await loadConfigsWithPassword(configPassword);
    
  } catch (error) {
    console.error('Failed to add config:', error);
    showErrorInModal(error.message);
  } finally {
    const submitBtn = event.target.querySelector('button[type="submit"]');
    submitBtn.disabled = false;
    submitBtn.textContent = '追加';
  }
}

// 型に応じた値変換
function convertValueByType(value, type) {
  const trimmedValue = value.trim();
  
  switch (type) {
    case 'boolean':
      const lowerValue = trimmedValue.toLowerCase();
      if (lowerValue === 'true' || lowerValue === '1') {
        return 'true';
      } else if (lowerValue === 'false' || lowerValue === '0') {
        return 'false';
      } else {
        throw new Error('Invalid boolean value. Use true/false or 1/0.');
      }
    case 'number':
      const num = parseFloat(trimmedValue);
      if (isNaN(num)) throw new Error('Invalid number value.');
      return num.toString();
    case 'json':
      try {
        const parsed = JSON.parse(trimmedValue);
        return JSON.stringify(parsed); // Return as JSON string
      } catch {
        throw new Error('Invalid JSON format.');
      }
    default:
      return trimmedValue;
  }
}

// イベントリスナーの追加
document.addEventListener('DOMContentLoaded', function() {
  // 既存のイベントリスナー...
  
  // 追加ボタン
  const addConfigBtn = document.getElementById('add-config-btn');
  if (addConfigBtn) {
    addConfigBtn.addEventListener('click', showAddConfigModal);
  }
  
  // モーダル関連
  const modalCloseBtn = document.getElementById('modal-close-btn');
  const cancelBtn = document.getElementById('cancel-btn');
  const addConfigForm = document.getElementById('add-config-form');
  
  if (modalCloseBtn) {
    modalCloseBtn.addEventListener('click', hideAddConfigModal);
  }
  
  if (cancelBtn) {
    cancelBtn.addEventListener('click', hideAddConfigModal);
  }
  
  if (addConfigForm) {
    addConfigForm.addEventListener('submit', handleAddConfig);
  }
  
  // モーダル外クリックで閉じる
  const modal = document.getElementById('add-config-modal');
  if (modal) {
    window.addEventListener('click', function(event) {
      if (event.target === modal) {
        hideAddConfigModal();
      }
    });
  }
});
```
