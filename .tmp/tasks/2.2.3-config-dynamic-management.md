# 2.2.3 設定管理UI - 動的設定項目管理

## ステータス
- [ ] 完了 / [x] 未完了

- 完了日: 未完了

## 概要
設定管理UIに動的な設定項目の追加・削除機能を実装する。管理者が標準設定項目以外にも、カスタム設定項目を柔軟に追加・編集・削除できる機能を提供し、アプリケーションの拡張性を向上させる。このシステムは既存のバリデーションシステム（2.2.2）と連携し、新しく追加される設定項目に対してもリアルタイムバリデーションを適用する。

## 機能要件

### 設定項目追加機能
- **目的**: 新しい設定項目の動的追加
- **機能**:
  - カスタム設定キーの追加
  - データ型の指定（string, boolean, number, json）
  - 説明文の設定
  - デフォルト値の設定

### 設定項目削除機能
- **目的**: 不要な設定項目の削除
- **機能**:
  - カスタム設定項目の削除
  - システム設定項目の保護
  - 削除確認ダイアログ
  - 依存関係チェック

### 設定項目編集機能
- **目的**: 設定項目のメタデータ編集
- **機能**:
  - 説明文の変更
  - データ型の変更
  - バリデーションルールの追加

```typescript
// src/services/config/dynamic.ts
export class DynamicConfigService {
  /**
   * Add new configuration item with validation rules
   */
  static async addConfig(
    key: string,
    value: string,
    type: ConfigType,
    options: {
      description?: string;
      systemConfig?: boolean;
      validationRules?: ValidationRule;
    } = {}
  ): Promise<Config> {
    // Validate key format
    if (!this.isValidConfigKey(key)) {
      throw new Error('Invalid configuration key format');
    }

    // Check for existing key
    if (ConfigService.has(key)) {
      throw new Error(`Configuration key '${key}' already exists`);
    }

    // Create configuration with validation rules
    const configData: ConfigInsert = {
      key,
      value,
      type,
      description: options.description,
      system_config: options.systemConfig ? 1 : 0,
      validation: options.validationRules ? JSON.stringify(options.validationRules) : null
    };

    // Use ConfigService to maintain cache consistency
    const result = await ConfigService.upsertWithValidation(
      key,
      value,
      type,
      options.description,
      options.validationRules
    );

    return result;
  }

  /**
   * Delete configuration item
   */
  static async deleteConfig(key: string): Promise<boolean> {
    // Check if it's a system configuration
    const config = ConfigService.findByKey(key);
    if (!config) {
      throw new Error(`Configuration key '${key}' not found`);
    }

    if (config.system_config) {
      throw new Error('Cannot delete system configuration items');
    }

    // Check dependencies before deletion
    await this.checkDependencies(key);

    // Delete from database and cache
    return await ConfigService.deleteByKey(key);
  }

  /**
   * Update configuration metadata
   */
  static async updateConfigMeta(
    key: string,
    metadata: {
      description?: string;
      type?: ConfigType;
      validationRules?: ValidationRule;
    }
  ): Promise<Config> {
    const existingConfig = ConfigService.findByKey(key);
    if (!existingConfig) {
      throw new Error(`Configuration key '${key}' not found`);
    }

    // Handle type conversion if needed
    let newValue = existingConfig.value;
    if (metadata.type && metadata.type !== existingConfig.type) {
      newValue = this.convertValue(existingConfig.value, existingConfig.type as ConfigType, metadata.type);
    }

    // Update configuration
    return await ConfigService.upsertWithValidation(
      key,
      newValue,
      metadata.type || existingConfig.type as ConfigType,
      metadata.description !== undefined ? metadata.description : existingConfig.description || undefined,
      metadata.validationRules
    );
  }

  /**
   * Get all custom (non-system) configurations
   */
  static getCustomConfigs(): Record<string, ConfigWithValidation> {
    const allConfigs = ConfigService.getAllWithValidation();
    const customConfigs: Record<string, ConfigWithValidation> = {};

    for (const [key, config] of Object.entries(allConfigs)) {
      if (!config.system_config) {
        customConfigs[key] = config;
      }
    }

    return customConfigs;
  }

  /**
   * Validate configuration key format
   */
  private static isValidConfigKey(key: string): boolean {
    // Must start with letter, contain only letters, numbers, dots, underscores
    return /^[a-zA-Z][a-zA-Z0-9._]*$/.test(key);
  }

  /**
   * Check dependencies before deletion
   */
  private static async checkDependencies(key: string): Promise<void> {
    // Implementation would check if other configurations depend on this key
    // For now, this is a placeholder for future dependency tracking
    
    // Example dependency checks:
    // - If storage.type is being deleted, check for storage.r2.* or storage.gdrive.* configs
    // - If auth provider config is being deleted, check for related auth configs
    
    // This could be implemented with a dependency map or by analyzing validation rules
  }

  /**
   * Convert value between types
   */
  private static convertValue(value: string, fromType: ConfigType, toType: ConfigType): string {
    if (fromType === toType) return value;

    try {
      switch (toType) {
        case 'string':
          return value;
        case 'boolean':
          if (fromType === 'string') {
            const lowerValue = value.toLowerCase();
            if (lowerValue === 'true' || lowerValue === '1' || lowerValue === 'yes') return 'true';
            if (lowerValue === 'false' || lowerValue === '0' || lowerValue === 'no') return 'false';
          }
          throw new Error(`Cannot convert "${value}" to boolean`);
        case 'number':
          const num = parseFloat(value);
          if (isNaN(num)) throw new Error(`Cannot convert "${value}" to number`);
          return num.toString();
        default:
          return value;
      }
    } catch (error) {
      throw new Error(`Type conversion failed: ${error.message}`);
    }
  }
}
```

```typescript
// src/api/v1/config/add.ts
export const addConfigHandler = async (c: Context<{ Bindings: Env }>) => {
  try {
    // Authentication check
    const isAuthenticated = await checkConfigAuthentication(c);
    if (!isAuthenticated) {
      return c.json({ error: { code: 'UNAUTHORIZED', message: 'Authentication required' } }, 401);
    }

    // Parse and validate request
    const { key, value, type, description, systemConfig, validationRules } = await c.req.json();
    
    // Input validation
    const validationResult = validateAddConfigRequest({ key, value, type, description, systemConfig, validationRules });
    if (!validationResult.valid) {
      return c.json({ 
        error: { code: 'VALIDATION_ERROR', message: validationResult.message, details: validationResult.errors } 
      }, 400);
    }

    // Add configuration using DynamicConfigService
    const newConfig = await DynamicConfigService.addConfig(key, value, type, {
      description,
      systemConfig,
      validationRules
    });

    return c.json({
      success: true,
      message: 'Configuration item added successfully',
      config: {
        key: newConfig.key,
        value: newConfig.value,
        type: newConfig.type,
        description: newConfig.description,
        systemConfig: Boolean(newConfig.system_config)
      }
    });

  } catch (error) {
    console.error('Add config error:', error);
    
    if (error.message.includes('already exists')) {
      return c.json({ error: { code: 'CONFLICT', message: error.message } }, 409);
    }
    
    return c.json({ error: { code: 'INTERNAL_ERROR', message: 'Failed to add configuration' } }, 500);
  }
};

// src/api/v1/config/delete.ts
export const deleteConfigHandler = async (c: Context<{ Bindings: Env }>) => {
  try {
    // Authentication check
    const isAuthenticated = await checkConfigAuthentication(c);
    if (!isAuthenticated) {
      return c.json({ error: { code: 'UNAUTHORIZED', message: 'Authentication required' } }, 401);
    }

    const key = c.req.param('key');
    if (!key) {
      return c.json({ error: { code: 'BAD_REQUEST', message: 'Configuration key is required' } }, 400);
    }

    // Delete configuration using DynamicConfigService
    const deleted = await DynamicConfigService.deleteConfig(key);

    if (deleted) {
      return c.json({
        success: true,
        message: 'Configuration item deleted successfully',
        deletedKey: key
      });
    } else {
      return c.json({ error: { code: 'NOT_FOUND', message: 'Configuration not found' } }, 404);
    }

  } catch (error) {
    console.error('Delete config error:', error);
    
    if (error.message.includes('Cannot delete system')) {
      return c.json({ error: { code: 'FORBIDDEN', message: error.message } }, 403);
    }
    
    if (error.message.includes('not found')) {
      return c.json({ error: { code: 'NOT_FOUND', message: error.message } }, 404);
    }
    
    return c.json({ error: { code: 'INTERNAL_ERROR', message: 'Failed to delete configuration' } }, 500);
  }
};
```

### 設定項目追加機能
```javascript
class ConfigManager {
  constructor() {
    this.setupEventListeners();
  }

  setupEventListeners() {
    // 設定項目追加フォーム
    document.getElementById('add-config-form').addEventListener('submit', (e) => {
      e.preventDefault();
      this.addConfig();
    });

    // データ型変更時の処理
    document.getElementById('new-config-type').addEventListener('change', (e) => {
      this.updateValueInput(e.target.value);
    });
  }

  // データ型に応じた入力フィールドの更新
  updateValueInput(type) {
    const valueInput = document.getElementById('new-config-value');
    if (!valueInput) return;

    // 入力タイプを変更
    switch (type) {
      case 'boolean':
        valueInput.type = 'checkbox';
        valueInput.placeholder = '';
        break;
      case 'number':
        valueInput.type = 'number';
        valueInput.placeholder = '数値を入力';
        break;
      case 'string':
      default:
        valueInput.type = 'text';
        valueInput.placeholder = '文字列を入力';
        break;
    }
  }

  // 個別設定の保存
  async saveConfig(key) {
    const input = document.querySelector(`input[name="${key}"], select[name="${key}"]`);
    if (!input) {
      this.showNotification(false, `設定項目 ${key} が見つかりません`);
      return;
    }

    const value = input.type === 'checkbox' ? input.checked.toString() : input.value;
    
    try {
      const response = await fetch('/api/v1/config/update', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'X-Requested-With': 'XMLHttpRequest'
        },
        body: JSON.stringify({
          updates: [{
            key: key,
            value: value,
            type: input.dataset.type || 'string'
          }]
        })
      });

      const result = await response.json();
      
      if (result.success) {
        this.showNotification(true, '設定が正常に更新されました');
      } else {
        this.showNotification(false, result.message);
      }
    } catch (error) {
      this.showNotification(false, `エラー: ${error.message}`);
    }
  }

  // 設定項目追加
  async addConfig() {
    const form = document.getElementById('add-config-form');
    const formData = new FormData(form);
    
    const configData = {
      key: formData.get('key'),
      value: formData.get('value'),
      type: formData.get('type'),
      description: formData.get('description'),
      systemConfig: formData.get('system_config') === 'on'
    };

    // クライアントサイドバリデーション
    if (!this.validateNewConfig(configData)) {
      return;
    }

    try {
      const response = await fetch('/api/v1/config/add', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'X-Requested-With': 'XMLHttpRequest'
        },
        body: JSON.stringify(configData)
      });

      const result = await response.json();
      
      if (result.success) {
        this.addConfigToTable(result.config);
        this.resetAddForm();
        this.showNotification(true, '設定項目が追加されました');
      } else {
        this.showNotification(false, result.message);
      }
    } catch (error) {
      this.showNotification(false, `エラー: ${error.message}`);
    }
  }

  // 設定項目削除
  async deleteConfig(key) {
    const isSystemConfig = document.querySelector(`tr[data-key="${key}"]`).dataset.system === 'true';
    
    if (isSystemConfig) {
      this.showNotification(false, 'System configuration items cannot be deleted');
      return;
    }

    if (!confirm(`Delete configuration item "${key}"?\nThis action cannot be undone.`)) {
      return;
    }

    try {
      const response = await fetch(`/api/v1/config/${encodeURIComponent(key)}`, {
        method: 'DELETE',
        headers: {
          'X-Requested-With': 'XMLHttpRequest'
        }
      });

      const result = await response.json();
      
      if (result.success) {
        this.removeConfigFromTable(key);
        this.showNotification(true, '設定項目が削除されました');
      } else {
        this.showNotification(false, result.message);
      }
    } catch (error) {
      this.showNotification(false, `エラー: ${error.message}`);
    }
  }

  // 設定項目メタデータ編集
  async editConfigMeta(key) {
    try {
      // 現在のメタデータを取得
      const response = await fetch(`/api/v1/config/${encodeURIComponent(key)}/meta`);
      const metadata = await response.json();
      
      if (metadata.success) {
        this.openEditModal(key, metadata.data);
      } else {
        this.showNotification(false, 'メタデータの取得に失敗しました');
      }
    } catch (error) {
      this.showNotification(false, `エラー: ${error.message}`);
    }
  }

  // 編集モーダルを開く
  openEditModal(key, metadata) {
    const modal = document.getElementById('edit-config-modal');
    
    // フォームに現在の値を設定
    document.getElementById('edit-config-key').value = key;
    document.getElementById('edit-config-description').value = metadata.description || '';
    document.getElementById('edit-config-type').value = metadata.type || 'string';
    
    // バリデーションルールの設定
    if (metadata.validationRules) {
      this.populateValidationRules(metadata.validationRules);
    }
    
    modal.style.display = 'block';
  }

  // テーブルに新しい設定項目を追加
  addConfigToTable(config) {
    const tbody = document.querySelector('.config-table tbody');
    const row = document.createElement('tr');
    row.dataset.key = config.key;
    row.dataset.system = config.systemConfig;
    
    const inputType = this.getInputType(config.type, config.key, config.value);
    const badge = config.systemConfig ? 
      '<span class="system-badge">システム</span>' : 
      '<span class="custom-badge">カスタム</span>';
    
    const deleteButton = config.systemConfig ? '' : 
      `<button type="button" onclick="configManager.deleteConfig('${config.key}')" class="btn-delete">削除</button>`;
    
    row.innerHTML = `
      <td>${config.key}</td>
      <td>${inputType}</td>
      <td>${config.type}</td>
      <td>${config.description || ''}</td>
      <td>${badge}</td>
      <td>
        <button type="button" onclick="configManager.saveConfig('${config.key}')" class="btn-save">保存</button>
        <button type="button" onclick="configManager.editConfigMeta('${config.key}')" class="btn-edit">編集</button>
        ${deleteButton}
      </td>
    `;
    
    tbody.appendChild(row);
  }

  // テーブルから設定項目を削除
  removeConfigFromTable(key) {
    const row = document.querySelector(`tr[data-key="${key}"]`);
    if (row) {
      row.remove();
    }
  }

  // 入力フィールドの種類を決定
  getInputType(type, key, value = '') {
    switch (type) {
      case 'boolean':
        const checked = value === 'true' || value === true ? 'checked' : '';
        return `<input type="checkbox" name="${key}" class="config-checkbox" data-type="boolean" ${checked}>`;
      case 'number':
        return `<input type="number" name="${key}" value="${value}" class="config-input" data-type="number">`;
      default:
        return `<input type="text" name="${key}" value="${value}" class="config-input" data-type="string">`;
    }
  }

  // 新規設定項目のバリデーション
  validateNewConfig(config) {
    if (!config.key || !config.key.match(/^[a-zA-Z][a-zA-Z0-9._]*$/)) {
      this.showNotification(false, '設定キーは英字で始まり、英数字・ドット・アンダースコアのみ使用可能です');
      return false;
    }

    if (!config.type) {
      this.showNotification(false, 'データ型を選択してください');
      return false;
    }

    // 既存キーの重複チェック
    if (document.querySelector(`tr[data-key="${config.key}"]`)) {
      this.showNotification(false, '同じキーの設定項目が既に存在します');
      return false;
    }

    return true;
  }

  // フォームリセット
  resetAddForm() {
    document.getElementById('add-config-form').reset();
  }

  // 通知表示
  showNotification(success, message) {
    // 既存の通知システムを使用
    const notification = document.getElementById('notification') || this.createNotification();
    notification.className = `notification ${success ? 'success' : 'error'}`;
    notification.innerHTML = `
      <span class="icon">${success ? '✅' : '❌'}</span>
      <span class="message">${message}</span>
      <button onclick="this.parentElement.style.display='none'" class="btn-close">×</button>
    `;
    notification.style.display = 'block';
    
    // 3秒後に自動で非表示
    setTimeout(() => {
      notification.style.display = 'none';
    }, 3000);
  }

  // 通知要素を作成
  createNotification() {
    const notification = document.createElement('div');
    notification.id = 'notification';
    notification.className = 'notification';
    notification.style.display = 'none';
    document.body.appendChild(notification);
    return notification;
  }
}

// グローバルインスタンス
const configManager = new ConfigManager();
```

```javascript
// static/js/config-dynamic-validator.js
class DynamicConfigValidator extends ConfigFormValidator {
  constructor(validationRules) {
    super(validationRules);
    this.dynamicRules = new Map();
  }

  // 動的に追加された設定項目のバリデーションルールを登録
  registerDynamicRule(key, validationRule) {
    this.dynamicRules.set(key, validationRule);
    this.rules[key] = validationRule;
  }

  // 設定項目削除時のバリデーションルール削除
  removeDynamicRule(key) {
    this.dynamicRules.delete(key);
    delete this.rules[key];
  }

  // 新規設定項目追加時のリアルタイムバリデーション有効化
  enableValidationForNewField(key) {
    const input = document.querySelector(`input[name="${key}"], select[name="${key}"]`);
    if (!input) return;

    // バリデーションクラスを追加
    input.classList.add('config-input');

    // イベントリスナーを追加
    input.addEventListener('input', (e) => this.validateField(e.target));
    input.addEventListener('focus', (e) => this.clearFieldError(e.target));
    input.addEventListener('blur', (e) => this.validateField(e.target));
  }

  // バリデーションルール動的更新
  async updateValidationRules() {
    try {
      const response = await fetch('/api/v1/config/validation-rules');
      if (response.ok) {
        const newRules = await response.json();
        
        // 既存ルールと動的ルールをマージ
        this.rules = { ...newRules };
        
        // 動的ルールを追加
        for (const [key, rule] of this.dynamicRules) {
          this.rules[key] = rule;
        }
      }
    } catch (error) {
      console.warn('Failed to update validation rules:', error);
    }
  }
}
```
