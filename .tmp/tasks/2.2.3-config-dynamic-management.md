# 2.2.3 設定管理UI - 動的設定項目管理

- [ ] 完了 / [x] 未完了

- 完了日: 未完了

## 概要
設定管理UIに動的な設定項目の追加・削除機能を実装する。管理者が標準設定項目以外にも、カスタム設定項目を柔軟に追加・編集・削除できる機能を提供し、アプリケーションの拡張性を向上させる。

## 機能要件

### 設定項目追加機能
- **目的**: 新しい設定項目の動的追加
- **機能**:
  - カスタム設定キーの追加
  - データ型の指定（string, boolean, number）
  - 説明文の設定
  - デフォルト値の設定

### 設定項目削除機能
- **目的**: 不要な設定項目の削除
- **機能**:
  - カスタム設定項目の削除
  - システム設定項目の保護
  - 削除確認ダイアログ
  - 依存関係チェック

### 設定項目編集機能
- **目的**: 設定項目のメタデータ編集
- **機能**:
  - 説明文の変更
  - データ型の変更
  - バリデーションルールの追加

interface AddConfigRequest {
  key: string;
  value: string;
  type: 'string' | 'boolean' | 'number';

  systemConfig?: boolean;
  validationRules?: {
    required?: boolean;
    minLength?: number;
    maxLength?: number;
    pattern?: string;
    minValue?: number;
    maxValue?: number;
  };

interface AddConfigResponse {
  success: boolean;
  message: string;
  config?: {
    key: string;
    value: string | boolean | number;
    type: string;
    description: string;
    systemConfig: boolean;
  };
}

## JavaScript機能実装

### 設定項目追加機能
```javascript
class ConfigManager {
  constructor() {
    this.setupEventListeners();
  }

  setupEventListeners() {
    // 設定項目追加フォーム
    document.getElementById('add-config-form').addEventListener('submit', (e) => {
      e.preventDefault();
      this.addConfig();
    });

    // データ型変更時の処理
    document.getElementById('new-config-type').addEventListener('change', (e) => {
      this.updateValueInput(e.target.value);
    });
  }

  // 個別設定の保存
  async saveConfig(key) {
    const input = document.querySelector(`input[name="${key}"], select[name="${key}"]`);
    if (!input) {
      this.showNotification(false, `設定項目 ${key} が見つかりません`);
      return;
    }

    const value = input.type === 'checkbox' ? input.checked.toString() : input.value;
    
    try {
      const response = await fetch('/api/v1/config/update', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'X-Requested-With': 'XMLHttpRequest'
        },
        body: JSON.stringify({
          updates: [{
            key: key,
            value: value,
            type: input.dataset.type || 'string'
          }]
        })
      });

      const result = await response.json();
      
      if (result.success) {
        this.showNotification(true, '設定が正常に更新されました');
      } else {
        this.showNotification(false, result.message);
      }
    } catch (error) {
      this.showNotification(false, `エラー: ${error.message}`);
    }
  }

  // 設定項目追加
  async addConfig() {
    const form = document.getElementById('add-config-form');
    const formData = new FormData(form);
    
    const configData = {
      key: formData.get('key'),
      value: formData.get('value'),
      type: formData.get('type'),
      description: formData.get('description'),
      systemConfig: formData.get('system_config') === 'on'
    };

    // クライアントサイドバリデーション
    if (!this.validateNewConfig(configData)) {
      return;
    }

    try {
      const response = await fetch('/api/v1/config/add', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'X-Requested-With': 'XMLHttpRequest'
        },
        body: JSON.stringify(configData)
      });

      const result = await response.json();
      
      if (result.success) {
        this.addConfigToTable(result.config);
        this.resetAddForm();
        this.showNotification(true, '設定項目が追加されました');
      } else {
        this.showNotification(false, result.message);
      }
    } catch (error) {
      this.showNotification(false, `エラー: ${error.message}`);
    }
  }

  // 設定項目削除
  async deleteConfig(key) {
    const isSystemConfig = document.querySelector(`tr[data-key="${key}"]`).dataset.system === 'true';
    
    if (isSystemConfig) {
      this.showNotification(false, 'System configuration items cannot be deleted');
      return;
    }

    if (!confirm(`Delete configuration item "${key}"?\nThis action cannot be undone.`)) {
      return;
    }

    try {
      const response = await fetch(`/api/v1/config/${encodeURIComponent(key)}`, {
        method: 'DELETE',
        headers: {
          'X-Requested-With': 'XMLHttpRequest'
        }
      });

      const result = await response.json();
      
      if (result.success) {
        this.removeConfigFromTable(key);
        this.showNotification(true, '設定項目が削除されました');
      } else {
        this.showNotification(false, result.message);
      }
    } catch (error) {
      this.showNotification(false, `エラー: ${error.message}`);
    }
  }

  // 設定項目メタデータ編集
  async editConfigMeta(key) {
    try {
      // 現在のメタデータを取得
      const response = await fetch(`/api/v1/config/${encodeURIComponent(key)}/meta`);
      const metadata = await response.json();
      
      if (metadata.success) {
        this.openEditModal(key, metadata.data);
      } else {
        this.showNotification(false, 'メタデータの取得に失敗しました');
      }
    } catch (error) {
      this.showNotification(false, `エラー: ${error.message}`);
    }
  }

  // 編集モーダルを開く
  openEditModal(key, metadata) {
    const modal = document.getElementById('edit-config-modal');
    
    // フォームに現在の値を設定
    document.getElementById('edit-config-key').value = key;
    document.getElementById('edit-config-description').value = metadata.description || '';
    document.getElementById('edit-config-type').value = metadata.type || 'string';
    
    // バリデーションルールの設定
    if (metadata.validationRules) {
      this.populateValidationRules(metadata.validationRules);
    }
    
    modal.style.display = 'block';
  }

  // テーブルに新しい設定項目を追加
  addConfigToTable(config) {
    const tbody = document.querySelector('.config-table tbody');
    const row = document.createElement('tr');
    row.dataset.key = config.key;
    row.dataset.system = config.systemConfig;
    
    const inputType = this.getInputType(config.type);
    const badge = config.systemConfig ? 
      '<span class="system-badge">システム</span>' : 
      '<span class="custom-badge">カスタム</span>';
    
    const deleteButton = config.systemConfig ? '' : 
      `<button type="button" onclick="configManager.deleteConfig('${config.key}')" class="btn-delete">削除</button>`;
    
    row.innerHTML = `
      <td>${config.key}</td>
      <td>${inputType}</td>
      <td>${config.type}</td>
      <td>${config.description || ''}</td>
      <td>${badge}</td>
      <td>
        <button type="button" onclick="configManager.saveConfig('${config.key}')" class="btn-save">保存</button>
        <button type="button" onclick="configManager.editConfigMeta('${config.key}')" class="btn-edit">編集</button>
        ${deleteButton}
      </td>
    `;
    
    tbody.appendChild(row);
  }

  // テーブルから設定項目を削除
  removeConfigFromTable(key) {
    const row = document.querySelector(`tr[data-key="${key}"]`);
    if (row) {
      row.remove();
    }
  }

  // 入力フィールドの種類を決定
  getInputType(type) {
    switch (type) {
      case 'boolean':
        return '<input type="checkbox" class="config-checkbox">';
      case 'number':
        return '<input type="number" class="config-input">';
      default:
        return '<input type="text" class="config-input">';
    }
  }

  // 新規設定項目のバリデーション
  validateNewConfig(config) {
    if (!config.key || !config.key.match(/^[a-zA-Z][a-zA-Z0-9._]*$/)) {
      this.showNotification(false, '設定キーは英字で始まり、英数字・ドット・アンダースコアのみ使用可能です');
      return false;
    }

    if (!config.type) {
      this.showNotification(false, 'データ型を選択してください');
      return false;
    }

    // 既存キーの重複チェック
    if (document.querySelector(`tr[data-key="${config.key}"]`)) {
      this.showNotification(false, '同じキーの設定項目が既に存在します');
      return false;
    }

    return true;
  }

  // フォームリセット
  resetAddForm() {
    document.getElementById('add-config-form').reset();
  }

  // 通知表示
  showNotification(success, message) {
    // 既存の通知システムを使用
    const notification = document.getElementById('notification') || this.createNotification();
    notification.className = `notification ${success ? 'success' : 'error'}`;
    notification.innerHTML = `
      <span class="icon">${success ? '✅' : '❌'}</span>
      <span class="message">${message}</span>
      <button onclick="this.parentElement.style.display='none'" class="btn-close">×</button>
    `;
    notification.style.display = 'block';
    
    // 3秒後に自動で非表示
    setTimeout(() => {
      notification.style.display = 'none';
    }, 3000);
  }

  // 通知要素を作成
  createNotification() {
    const notification = document.createElement('div');
    notification.id = 'notification';
    notification.className = 'notification';
    notification.style.display = 'none';
    document.body.appendChild(notification);
    return notification;
  }
}

// グローバルインスタンス
const configManager = new ConfigManager();
```
