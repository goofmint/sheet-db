## 概要
設定管理UIの設定値バリデーションシステムを実装する。各設定項目に対して適切な検証ルールを適用し、無効な設定値の保存を防ぐとともに、ユーザーにわかりやすいエラーメッセージを提供する。

## 機能要件

### バリデーション機能
- **目的**: 設定値の整合性と有効性を保証
- **機能**:
  - フィールド別バリデーションルール
  - リアルタイムバリデーション
  - 依存関係バリデーション
  - カスタムバリデーター対応

### エラー表示機能
- **目的**: ユーザビリティの向上
- **機能**:
  - インラインエラーメッセージ
  - 日本語エラーメッセージ
  - フィールドハイライト
  - エラーサマリー表示

  errorMessage: string;
  default?: string | number | boolean;

### ConfigServiceの拡張
```typescript
// ConfigServiceにバリデーション関連メソッドを追加
interface ConfigWithValidation {
  key: string;
  value: string;
  type: ConfigType;
  description: string | null;
  validation: ValidationRule | null;
  system_config: boolean;
}

// ConfigServiceに追加するメソッド
class ConfigService {
  // バリデーションルール付きで設定を取得
  static async getWithValidation(key: string): Promise<ConfigWithValidation | null> {
    const config = await this.get(key);
    if (!config) return null;
    
    const validation = config.validation ? JSON.parse(config.validation) : null;
    return {
      ...config,
      validation,
      system_config: config.system_config || false
    };
  }

  // 全設定をバリデーションルール付きで取得
  static async getAllWithValidation(): Promise<Record<string, ConfigWithValidation>> {
    const configs = await this.getAll();
    const result: Record<string, ConfigWithValidation> = {};
    
    for (const [key, config] of Object.entries(configs)) {
      const validation = config.validation ? JSON.parse(config.validation) : null;
      result[key] = {
        ...config,
        validation,
        system_config: config.system_config || false
      };
    }
    
    return result;
  }

  // バリデーションルールを設定
  static async setValidation(key: string, validation: ValidationRule): Promise<void> {
    await this.update(key, {
      validation: JSON.stringify(validation)
    });
  }
}

export abstract class BaseValidator {
  abstract validate(value: string, rule: ValidationRule): ValidationResult;
  
  protected createError(message: string): ValidationResult {
    return { valid: false, error: message };
  }
  
  protected createSuccess(): ValidationResult {
    return { valid: true };
  }
}
```

#### 文字列バリデーター
```typescript
// src/utils/validators/string-validator.ts
import { BaseValidator } from './base-validator';

export class StringValidator extends BaseValidator {
  validate(value: string, rule: ValidationRule): ValidationResult {
    // Required check
    if (rule.required && (!value || value.trim() === '')) {
      return this.createError('This field is required');
    }
    
    // If empty and not required, it's valid
    if (!value || value.trim() === '') {
      return this.createSuccess();
    }
    
    // Length validation
    if (rule.minLength && value.length < rule.minLength) {
      return this.createError(rule.errorMessage);
    }
    
    if (rule.maxLength && value.length > rule.maxLength) {
      return this.createError(rule.errorMessage);
    }
    
    // Pattern validation
    if (rule.pattern) {
      const regex = new RegExp(rule.pattern);
      if (!regex.test(value)) {
        return this.createError(rule.errorMessage);
      }
    }
    
    // Enum validation
    if (rule.enum && !rule.enum.includes(value)) {
      return this.createError(rule.errorMessage);
    }
    
    return this.createSuccess();
  }
}
```

// src/utils/validators/number-validator.ts
import { BaseValidator } from './base-validator';

export class NumberValidator extends BaseValidator {
  validate(value: string, rule: ValidationRule): ValidationResult {
    // Required check
    if (rule.required && (!value || value.trim() === '')) {
      return this.createError('This field is required');
    }
    
    // If empty and not required, it's valid
    if (!value || value.trim() === '') {
      return this.createSuccess();
    }
    
    // Type validation - check if it's a valid number
    const numValue = parseFloat(value);
    if (isNaN(numValue) || !isFinite(numValue)) {
      return this.createError(rule.errorMessage || 'Must be a valid number');
    }
    
    // Range validation
    if (rule.min !== undefined && numValue < rule.min) {
      return this.createError(rule.errorMessage);
    }
    
    if (rule.max !== undefined && numValue > rule.max) {
      return this.createError(rule.errorMessage);
    }
    
    return this.createSuccess();
  }
}

#### バリデーターファクトリー
```typescript
// src/utils/validators/validator-factory.ts
import { BaseValidator } from './base-validator';
import { StringValidator } from './string-validator';
import { NumberValidator } from './number-validator';
import { BooleanValidator } from './boolean-validator';

export class ValidatorFactory {
  private static validators: Map<string, BaseValidator> = new Map([
    ['string', new StringValidator()],
    ['number', new NumberValidator()],
    ['boolean', new BooleanValidator()]
  ]);
  
  static getValidator(type: string): BaseValidator {
    const validator = this.validators.get(type);
    if (!validator) {
      // Default to string validator for unknown types
      return this.validators.get('string')!;
    }
    return validator;
  }
  
  static registerValidator(type: string, validator: BaseValidator): void {
    this.validators.set(type, validator);
  }
}
```

### メイン設定バリデータークラス
```typescript
// src/utils/config-validator.ts
import { ValidatorFactory } from './validators/validator-factory';

export class ConfigValidator {
  private configService: typeof ConfigService;

  constructor(configService: typeof ConfigService) {
    this.configService = configService;
  }

  // Get validation rules dynamically from database
  private async getValidationRules(): Promise<Record<string, ValidationRule>> {
    const configs = await this.configService.getAllWithValidation();
    const rules: Record<string, ValidationRule> = {};
    
    for (const [key, config] of Object.entries(configs)) {
      if (config.validation) {
        rules[key] = config.validation;
      }
    }
    
    return rules;
  }

  // Single field validation (async version)
  async validateField(key: string, value: string, allConfig: Record<string, string>): Promise<ValidationResult> {
    const rules = await this.getValidationRules();
    const rule = rules[key];
    
    if (!rule) {
      return { valid: true };
    }

    // Get type-specific validator
    const configType = rule.type || 'string';
    const validator = ValidatorFactory.getValidator(configType);
    
    // Use type-specific validator
    const result = validator.validate(value, rule);
    
    // If basic validation failed, return the error
    if (!result.valid) {
      return result;
    }

    // Additional dependency validation for this specific field
    const dependencyErrors = this.validateFieldDependencies(key, value, allConfig);
    if (dependencyErrors.length > 0) {
      return {
        valid: false,
        error: dependencyErrors[0] // Return first dependency error
      };
    }

    return { valid: true };
  }
  
  // Field-specific dependency validation
  private validateFieldDependencies(key: string, value: string, config: Record<string, string>): string[] {
    const errors: string[] = [];
    
    // Storage type dependencies
    if (key === 'storage.r2.bucket' || key === 'storage.r2.access_key_id' || 
        key === 'storage.r2.secret_access_key' || key === 'storage.r2.endpoint') {
      if (config['storage.type'] !== 'r2' && value.trim() !== '') {
        errors.push('R2 configuration is only valid when storage type is r2');
      }
    }
    
    if (key === 'storage.gdrive.folder_id') {
      if (config['storage.type'] !== 'gdrive' && value.trim() !== '') {
        errors.push('Google Drive configuration is only valid when storage type is gdrive');
      }
    }
    
    return errors;
  }

  // Validate all configuration (async version)
  async validateAll(config: Record<string, string>): Promise<ValidationResults> {
    const rules = await this.getValidationRules();
    const results: ValidationResults = {
      valid: true,
      errors: {}
    };

    // Check all rules for existing values or missing required fields
    for (const ruleKey of Object.keys(rules)) {
      const value = config[ruleKey] || '';
      const result = await this.validateField(ruleKey, value, config);
      if (!result.valid) {
        results.valid = false;
        results.errors[ruleKey] = result.error!;
      }
    }

    // Validate additional config items (not defined in rules)
    for (const [key, value] of Object.entries(config)) {
      if (!rules[key]) {
        const result = await this.validateField(key, value, config);
        if (!result.valid) {
          results.valid = false;
          results.errors[key] = result.error!;
        }
      }
    }

    // Dependency validation
    const dependencyErrors = this.validateDependencies(config);
    if (Object.keys(dependencyErrors).length > 0) {
      results.valid = false;
      results.errors = { ...results.errors, ...dependencyErrors };
    }

    return results;
  }

  // Dependency validation
  private validateDependencies(config: Record<string, string>): Record<string, string> {
    const errors: Record<string, string> = {};

    // Storage configuration dependencies
    const storageType = config['storage.type'];
    if (storageType === 'r2') {
      const requiredR2Fields = [
        'storage.r2.bucket',
        'storage.r2.access_key_id',
        'storage.r2.secret_access_key',
        'storage.r2.endpoint'
      ];
      
      for (const field of requiredR2Fields) {
        if (!config[field] || config[field].trim() === '') {
          errors[field] = `${field} is required when using R2 storage`;
        }
      }
    } else if (storageType === 'gdrive') {
      if (!config['storage.gdrive.folder_id'] || config['storage.gdrive.folder_id'].trim() === '') {
        errors['storage.gdrive.folder_id'] = 'Folder ID is required when using Google Drive storage';
      }
    }

    // Auth0 and Google mutual dependencies
    const hasAuth0Config = config['auth0.domain'] && config['auth0.client_id'] && config['auth0.client_secret'];
    const hasGoogleConfig = config['google.client_id'] && config['google.client_secret'];

    if (!hasAuth0Config && !hasGoogleConfig) {
      errors['auth0.domain'] = 'Either Auth0 or Google configuration is required';
      errors['google.client_id'] = 'Either Auth0 or Google configuration is required';
    }

    return errors;
  }

}
```

// src/utils/validators/email-validator.ts
import { BaseValidator } from './base-validator';

export class EmailValidator extends BaseValidator {
  private emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  
  validate(value: string, rule: ValidationRule): ValidationResult {
    if (rule.required && (!value || value.trim() === '')) {
      return this.createError('Email is required');
    }
    
    if (!value || value.trim() === '') {
      return this.createSuccess();
    }
    
    if (!this.emailRegex.test(value)) {
      return this.createError(rule.errorMessage || 'Must be a valid email address');
    }
    
    return this.createSuccess();
  }
}

## フロントエンドバリデーション

### リアルタイムバリデーション
```javascript
class ConfigFormValidator {
  constructor(validationRules) {
    this.rules = validationRules;
    this.setupEventListeners();
  }

  // イベントリスナーの設定
  setupEventListeners() {
    const configInputs = document.querySelectorAll('.config-input');
    
    configInputs.forEach(input => {
      // リアルタイムバリデーション（入力時）
      input.addEventListener('input', (e) => {
        this.validateField(e.target);
      });
      
      // フォーカス時の処理
      input.addEventListener('focus', (e) => {
        this.clearFieldError(e.target);
      });
      
      // フォーカス外し時の検証
      input.addEventListener('blur', (e) => {
        this.validateField(e.target);
      });
    });
  }

  // 単一フィールドの検証
  validateField(input) {
    const key = input.name;
    const value = input.value;
    const allConfig = this.getAllConfigValues();

    // バリデーション実行
    const result = this.validateSingleField(key, value, allConfig);
    
    if (result.valid) {
      this.showFieldSuccess(input);
    } else {
      this.showFieldError(input, result.error);
    }

    return result.valid;
  }

  // 全フィールドの検証
  validateAllFields() {
    const allConfig = this.getAllConfigValues();
    let isValid = true;
    const errors = {};

    for (const [key, value] of Object.entries(allConfig)) {
      const result = this.validateSingleField(key, value, allConfig);
      if (!result.valid) {
        isValid = false;
        errors[key] = result.error;
        
        const input = document.querySelector(`input[name="${key}"]`);
        if (input) {
          this.showFieldError(input, result.error);
        }
      }
    }

    // エラーサマリーの表示
    if (!isValid) {
      this.showErrorSummary(errors);
    } else {
      this.hideErrorSummary();
    }

    return isValid;
  }

  // フィールドエラーの表示
  showFieldError(input, message) {
    input.classList.add('error');
    
    // 既存のエラーメッセージを削除
    const existingError = input.parentNode.querySelector('.error-message');
    if (existingError) {
      existingError.remove();
    }

    // エラーメッセージを追加
    const errorElement = document.createElement('div');
    errorElement.className = 'error-message';
    errorElement.textContent = message;
    input.parentNode.appendChild(errorElement);
  }

  // フィールドエラーのクリア
  clearFieldError(input) {
    input.classList.remove('error', 'success');
    const errorMessage = input.parentNode.querySelector('.error-message');
    if (errorMessage) {
      errorMessage.remove();
    }
  }

  // フィールド成功の表示
  showFieldSuccess(input) {
    input.classList.remove('error');
    input.classList.add('success');
    
    const errorMessage = input.parentNode.querySelector('.error-message');
    if (errorMessage) {
      errorMessage.remove();
    }
  }

  // エラーサマリーの表示
  showErrorSummary(errors) {
    const errorCount = Object.keys(errors).length;
    const summaryElement = document.getElementById('error-summary') || this.createErrorSummary();
    
    summaryElement.innerHTML = `
      <div class="error-header">
        <span class="icon">⚠️</span>
        <strong>${errorCount}件のエラーがあります</strong>
      </div>
      <ul class="error-list">
        ${Object.entries(errors).map(([key, message]) => 
          `<li><strong>${key}:</strong> ${message}</li>`
        ).join('')}
      </ul>
    `;
    
    summaryElement.style.display = 'block';
    summaryElement.scrollIntoView({ behavior: 'smooth' });
  }

  // エラーサマリーの非表示
  hideErrorSummary() {
    const summaryElement = document.getElementById('error-summary');
    if (summaryElement) {
      summaryElement.style.display = 'none';
    }
  }

  // エラーサマリー要素の作成
  createErrorSummary() {
    const summaryElement = document.createElement('div');
    summaryElement.id = 'error-summary';
    summaryElement.className = 'error-summary';
    
    const form = document.getElementById('config-form');
    form.insertBefore(summaryElement, form.firstChild);
    
    return summaryElement;
  }

  // 全設定値の取得
  getAllConfigValues() {
    const inputs = document.querySelectorAll('.config-input');
    const config = {};
    
    inputs.forEach(input => {
      config[input.name] = input.value;
    });
    
    return config;
  }

  // 単一フィールドバリデーション（フロントエンド版）
  validateSingleField(key, value, allConfig) {
    // サーバーサイドのバリデーションロジックと同様の実装
    // 簡略化された例
    const rule = this.rules[key];
    if (!rule) return { valid: true };

    if (rule.required && (!value || value.trim() === '')) {
      return { valid: false, error: `${key}は必須項目です` };
    }

    if (rule.pattern && value) {
      const regex = new RegExp(rule.pattern);
      if (!regex.test(value)) {
        return { valid: false, error: rule.errorMessage };
      }
    }

    return { valid: true };
  }
}
```

import { ConfigValidator } from '../../utils/config-validator';

export const configUpdateHandler = async (c: Context) => {

      message: 'Validation error',
      errors: validationResult.errors

## メモ

### 実装方針
1. **データベース駆動設計**
   - バリデーションルールをConfigテーブルで管理
   - 動的なバリデーションルール変更対応
   - ハードコードの排除

2. **型別バリデーター分離**
   - 各データ型（string, number, boolean）ごとに専用バリデータークラス
   - 基底クラスによる共通処理の統一
   - ファクトリーパターンによる型別バリデーター管理

3. **拡張性重視**
   - 新しいデータ型のバリデーター追加が容易
   - ValidatorFactory.registerValidator()による動的登録
   - 基底クラス継承による一貫したインターフェース

4. **単一責任原則**
   - 各バリデータークラスは特定の型のみ担当
   - メインバリデータークラスは依存関係とルール管理に集中
   - ファクトリーはバリデーター選択のみに責任を持つ

5. **パフォーマンス考慮**
   - バリデーターインスタンスの再利用
   - データベースアクセスの最小化
   - 型判定によるバリデーター選択の高速化
